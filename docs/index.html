快速开始 — JavaScript API

import init, { PhysicsWorld } from './pkg/mps_wasm.js';

await init();
const world = new PhysicsWorld(0, -9.81, 0);

// 创建动态方块
const box = world.create_dynamic_box(0, 5, 0, 0.5, 0.5, 0.5, 1.0);
// 创建地面
const ground = world.create_ground_plane(0, 1, 0, 0);

// 每帧步进
function loop() {
  world.step(1/60);
  const pos = world.get_body_translation(box);
  // 同步到 Three.js mesh.position.set(pos.x, pos.y, pos.z)
  requestAnimationFrame(loop);
}

WASM API 参考

方法说明
new PhysicsWorld(gx, gy, gz)创建物理世界（设置重力）
world.step(dt)推进模拟 dt 秒
world.create_dynamic_box(px, py, pz, hx, hy, hz, mass) → u64创建动态方块，返回句柄
world.create_dynamic_sphere(px, py, pz, r, mass) → u64创建动态球体
world.create_ground_plane(nx, ny, nz, dist) → u64创建地面（半空间碰撞体）
world.insert_rigid_body(descriptor) → u64从描述符插入刚体
world.insert_collider(descriptor, parent) → u64插入碰撞体
world.get_body_translation(handle) → Vec3获取刚体位置
world.get_body_rotation(handle) → Quat获取刚体旋转（四元数）
world.get_body_linear_velocity(handle) → Vec3获取线速度
world.add_force(handle, fx, fy, fz, wake)施加力
world.apply_impulse(handle, ix, iy, iz, wake)施加冲量
world.register_celestial_gravity(bodyId, degree) → u64注册天体引力（0=Sun, 3=Earth, ...）
world.get_body_snapshot() → Float64Array批量获取所有刚体状态（13 f64/体）
world.get_collision_events() → Array获取碰撞事件
world.destroy()销毁世界释放内存
