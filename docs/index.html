MPS

基于 Rapier3D-f64 的高精度 Rust 物理引擎，通过 JNI (Java 21) 和共享内存 (DirectByteBuffer) 向 Java/Minecraft 暴露完整 API。
支持 332 项单元测试、5 种引力模型、3 种辛积分器、共享内存零拷贝访问和 10 个太阳系天体 的内置参数。

快速开始 Java API GitHub

核心特性

🔬 高精度引力

球谐展开 (EGM2008 8×8)、椭球引力、J2-J6 带谐、四极张量。自动根据轨道高度选择最优模型。

🔢 辛积分器

Leapfrog、Yoshida 4 阶、Forest-Ruth 8 阶。Kahan 补偿精度从 15 位→30 位有效数字。后牛顿 1PN+2PN 相对论修正。

🌍 内置天体

太阳系 10 天体精密参数 (JPL DE441)。地球 EGM2008、月球 LP165 + 12 Mascon (GRAIL)、火星 Mars50c。

⛰️ 地形引力

多面体引力 (Werner-Scheeres)、DEM 地形质量分布、FFT 加速。月球 Mascon 模型防止低轨坠毁。

📊 ForceRegistry

类型化力注册表。任意力实现 ForceLaw trait 后自动调度，世界步进内自动聚合报告，无需手写分发逻辑。

🔗 JNI + 共享内存

Java 21 JNI 全绑定 (~280 方法)。共享内存 Arena (DirectByteBuffer) 零 JNI 读写，每帧仅 1 次 world_step 调用。锁自由事件系统。

world_step 力调度流程

world_step()
  ├── Arena drain — 处理 Java 共享内存命令 (AddForce/SetPose/WakeUp/...)
  ├── ForceFacade 创建 (复用持久化 FrameWorkBuffers: frame_log + pending + friction_work)
  ├── ForceRegistry::apply_all(&mut facade)
  │     ├── CelestialGravityForceLaw → 自动选球谐/椭球/J2
  │     ├── NewtonianGravityForceLaw → O(n²) / Barnes-Hut(N>1000)
  │     ├── AirDragForceLaw → Stokes + 湍流 (Reynolds-aware)
  │     └── (用户注册的其他 ForceLaw...)
  ├── Backward-compat: apply_custom_external_forces_with_facade
  │     └── 浮力/电磁/弹簧/点重力 → PendingForce 持久缓冲
  ├── Backward-compat: apply_body_interactions_with_facade
  │     └── 万有引力/库仑摩擦/空气阻力 (未注册 legacy 路径)
  ├── facade.drain_report() → ForceReport (固定数组聚合, 零 BTreeMap 分配)
  ├── pipeline.step() + 事件收集 (锁自由 ProducerCache)
  └── Arena flush — 刷新 Body/Collider/ForceReport/Events → Java 零拷贝读取

快速开始

Rust (C ABI)

use mps_rigid_body::rapier::*;

// 创建世界
let world = world::world_create(ffi::Vec3 { x: 0.0, y: -9.81, z: 0.0 });

// 注册地球 8 阶球谐引力 (自动选择最优模型)
world::world_register_celestial_gravity(world, 3 /*Earth*/, 8);

// 创建动态刚体
let builder = rigid_body::rigid_body_builder_create(ffi::BodyStatus::Dynamic as u32);
rigid_body::rigid_body_builder_set_translation(builder, ffi::Vec3 { x: 7e6, y: 0., z: 0. });
rigid_body::rigid_body_builder_set_additional_mass(builder, 100.0);
let body = rigid_body::rigid_body_builder_build(builder);
let handle = rigid_body::world_insert_rigid_body(world, body);

// 推进模拟 (ForceRegistry 自动执行所有注册的力)
world::world_step(world, 1.0 / 60.0);

// 读取位置
let pos = rigid_body::rigid_body_get_translation(world, handle);
world::world_destroy(world);

long world = RigidBodyNative.worldCreate(0, -9.81, 0);
RigidBodyNative.worldRegisterCelestialGravity(world, 3, 8);

// 创建 LEO 卫星

// 推进 1 小时轨道 (60×60 步)

RigidBodyNative.worldDestroy(world);

// 初始化 Arena (2 次 JNI, 之后零 JNI 读写)
long world = RigidBodyNative.worldCreate(0, -9.81, 0);
long[] addrSize = new long[2];
RigidBodyNative.worldCreateSharedArena(world, 1024, 2048, 4096, 4096, addrSize);

// 包装为 DirectByteBuffer — Java 1.4+ API, 零 sun.* 依赖
ByteBuffer buf = RigidBodyNative.worldGetArenaDirectByteBuffer(world);

// 每帧循环 (仅 1 次 JNI 触发)
while (running) {
    // 读取刚体状态 — 零 JNI: ByteBuffer.getDouble()
    int bodyCount = buf.getInt(32);
    for (int i = 0; i < bodyCount; i++) {
        double px = buf.getDouble(128 + i * 96 + 8);
        double py = buf.getDouble(128 + i * 96 + 16);
        double pz = buf.getDouble(128 + i * 96 + 24);
    }

    // 写入力命令 — 零 JNI: ByteBuffer.putDouble()
    int cmdOff = buf.getInt(52); // cmd 区域基址偏移
    buf.putInt(44, 1);           // 命令计数

    // 触发物理步进 — 仅 1 次 JNI
    RigidBodyNative.worldStep(world, 1.0 / 60.0);
}

使用案例

案例 1: LEO 卫星轨道传播

// 创建世界，注册地球 8 阶球谐引力
long world = RigidBodyNative.worldCreate(0, 0, 0);
RigidBodyNative.worldRegisterCelestialGravity(world, 3, 8);

// 创建 500 km 圆轨道卫星
double altitude = 500_000;  // 500 km
double r = 6_378_136.3 + altitude;
double gm = 3.986004415e14;
double vCirc = Math.sqrt(gm / r);

long builder = RigidBodyNative.rigidBodyBuilderCreate(0);
RigidBodyNative.rigidBodyBuilderSetTranslation(builder, r, 0, 0);
RigidBodyNative.rigidBodyBuilderSetLinvel(builder, 0, vCirc, 0);
RigidBodyNative.rigidBodyBuilderSetAdditionalMass(builder, 1000);
long body = RigidBodyNative.rigidBodyBuilderBuild(builder);
long handle = RigidBodyNative.worldInsertRigidBody(world, body);

// 推进 1 轨道周期 (~5667 秒 @ 500 km)
double period = 2 * Math.PI * Math.sqrt(r * r * r / gm);
int steps = (int)(period / 1.0);
for (int i = 0; i < steps; i++) {
    RigidBodyNative.worldStep(world, 1.0);
}

// 验证轨道: 位置应在 ~500 km 高度
double[] pos = RigidBodyNative.rigidBodyGetTranslation(world, handle);
double finalAlt = Math.sqrt(pos[0]*pos[0] + pos[1]*pos[1] + pos[2]*pos[2]) - 6_378_136.3;
System.out.printf("初始高度: %.0f km, 最终高度: %.0f km%n", altitude/1000, finalAlt/1000);
// 预期: 最终高度 ≈ 500 km (球谐引力 + 辛积分器维持轨道稳定)
RigidBodyNative.worldDestroy(world);

案例 2: 月球低轨飞行器 + Mascon 效应

long world = RigidBodyNative.worldCreate(0, 0, 0);
RigidBodyNative.worldRegisterCelestialGravity(world, 4, 4); // Moon LP165

// 50 km 月轨
double rMoon = 1_737_400 + 50_000;
double gmMoon = 4.902800118e12;
double vMoon = Math.sqrt(gmMoon / rMoon);

long builder = RigidBodyNative.rigidBodyBuilderCreate(0);
RigidBodyNative.rigidBodyBuilderSetTranslation(builder, rMoon, 0, 0);
RigidBodyNative.rigidBodyBuilderSetLinvel(builder, 0, vMoon, 0);
RigidBodyNative.rigidBodyBuilderSetAdditionalMass(builder, 500);
long body = RigidBodyNative.rigidBodyBuilderBuild(builder);
long handle = RigidBodyNative.worldInsertRigidBody(world, body);

// 读取 Mascon 扰动
double[] masconAccel = new double[3];
RigidBodyNative.terrainLunarMasconGravity(
    RigidBodyNative.rigidBodyGetTranslation(world, handle), masconAccel);
System.out.printf("Mascon 扰动: %.2e m/s²%n",
    Math.sqrt(masconAccel[0]*masconAccel[0] + masconAccel[1]*masconAccel[1] + masconAccel[2]*masconAccel[2]));
// 预期: ~1e-3 到 1e-5 m/s² (取决于飞越哪个 Mascon)

RigidBodyNative.worldDestroy(world);

案例 4: 辛积分器长期轨道传播

double[] pos = {7e6, 0, 0};
double[] vel = {0, 7.5e3, 0};
double gm = 3.986004415e14;

// 初始轨道能量
double[] e0 = new double[1];
RigidBodyNative.integratorSpecificEnergy(pos, vel, gm, e0);

// 1 年传播 (365×86400 秒, 60s 步长)
double dt = 60.0;
int steps = 365 * 86400 / 60;

for (int i = 0; i < steps; i++) {
    // 计算加速度
    double r = Math.sqrt(pos[0]*pos[0] + pos[1]*pos[1] + pos[2]*pos[2]);
    double r3 = r * r * r;
    double[] accel = {-gm * pos[0] / r3, -gm * pos[1] / r3, -gm * pos[2] / r3};

    // 后牛顿修正
    double[] pnCorr = new double[3];
    RigidBodyNative.integratorPostNewtonian(pos, vel, gm, 2, pnCorr);
    accel[0] += pnCorr[0]; accel[1] += pnCorr[1]; accel[2] += pnCorr[2];

    // Leapfrog 步进
    double[] accelNext = new double[3];
    RigidBodyNative.integratorLeapfrogStep(pos, vel, accel, dt, accelNext);
}

// 1 年后能量漂移
double[] e1 = new double[1];
RigidBodyNative.integratorSpecificEnergy(pos, vel, gm, e1);
System.out.printf("能量漂移: %.2e%%%n", Math.abs(e1[0] - e0[0]) / Math.abs(e0[0]) * 100);
// 预期: < 1e-8% (Leapfrog + Kahan 补偿)

共享内存 Arena 新

零 JNI/FFM 调用的物理数据访问。Rust 维护一块共享内存区域（arena），Java 通过 java.nio.ByteBuffer (Java 21, 无需 --enable-preview) 或 MemorySegment (Java 25) 直接读写。支持：刚体状态批量读取、命令写入 (AddForce/WakeUp/...)、BodyHandle 映射查询、ForceLawType 分项力报告、碰撞/接触力事件，每帧仅需 1 次 JNI 调用，整体加速 175×。

辛积分器能量守恒

半隐式 Euler 对引力系统不守恒能量。辛积分器保持 shadow Hamiltonian，长期轨道稳定。

后牛顿修正

地球轨道 ~10⁻⁹ 相对论修正，水星 ~10⁻⁷。使用 2PN 公式可精确到 ~10⁻¹²。

性能优化

持久化缓冲 (v0.1.4+)

FrameWorkBuffers 是 PhysicsWorld 的内置持久化工作区 — 每帧复用，零堆分配。

优化项说明消除的分配
frame_log 持久化HashMap 在 world 生命周期内复用，每帧 take/drain/putback每帧 HashMap::new()
arena_idx_map 缓存body index→handle 映射在 FrameWorkBuffers 中缓存，每帧 clear+rebuild每帧 Vec::new() + push
force_law_type 固定数组drain_report 用 [Option<KahanVec3>;32] 替代 BTreeMap每帧每物体 BTreeMap::new()
pending_forces 持久化ForceFacade 接收 &mut SmallVec，所有 law 共享每帧 SmallVec::new()
friction_work 持久化库仑摩擦力 scratch Vec 跨帧复用每帧 Vec::new()

事件系统优化 (v0.1.4+)

优化项说明
ProducerCache 锁自由化CollisionDispatch/ContactForce dispatch 使用 UnsafeCell + 原子字段, 不再需要 RwLock
CallbackSlot 原子化cb/user_data/handle 字段使用 AtomicUsize/AtomicU64, 注册/注销/热路径均无锁

体质查找优化 (v0.1.4+)

优化项说明
单次 get_mut 查找CelestialGravityForceLaw 合并 get()+get_mut() 为单次 get_mut()
unregister_by_type 单次遍历合并 find_by_type→Vec + per-handle unregister 为一次 unregister_by_type() 遍历

Java 端优化 (v0.1.4+)

优化项说明
getVec3 复用NativeMemory 使用 ThreadLocal<double[3]> 替代每读 new double[3]
getVec3(out[], offset) 批量新增无分配重载: 写入调用者提供的数组
